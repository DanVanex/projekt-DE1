library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity safe_fsm is
    Port (
        clk       : in  STD_LOGIC;
        rst       : in  STD_LOGIC;
        btn_pulse : in  STD_LOGIC;
        sw_i      : in  STD_LOGIC_VECTOR(3 downto 0);
        digit_idx : out integer range 0 to 3;
        state_o   : out integer range 0 to 2 -- 0: INPUT, 1: OPEN, 2: ERR
    );
end safe_fsm;

architecture Behavioral of safe_fsm is
    type state_type is (S_INPUT, S_EVAL, S_OPEN, S_ERR);
    signal current_state : state_type := S_INPUT;
    type code_array is array (0 to 3) of unsigned(3 downto 0);
    constant SECRET : code_array := (x"1", x"2", x"3", x"4");
    signal entered : code_array := (others => x"0");
    signal cnt : integer range 0 to 3 := 0;
begin
    process(clk)
    begin
        if rising_edge(clk) then
            if rst = '1' then
                current_state <= S_INPUT;
                cnt <= 0;
            else
                case current_state is
                    when S_INPUT =>
                        if btn_pulse = '1' then
                            entered(cnt) <= unsigned(sw_i);
                            if cnt = 3 then current_state <= S_EVAL;
                            else cnt <= cnt + 1; end if;
                        end if;
                    when S_EVAL =>
                        if entered = SECRET then current_state <= S_OPEN;
                        else current_state <= S_ERR; end if;
                    when others => null; -- Zůstane v OPEN/ERR do resetu
                end case;
            end if;
        end if;
    end process;

    digit_idx <= cnt;
    state_o <= 0 when current_state = S_INPUT else
               1 when current_state = S_OPEN else 2;
end Behavioral;
