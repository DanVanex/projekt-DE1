library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity debouncer is
    Port (
        clk      : in  STD_LOGIC;
        rst      : in  STD_LOGIC;
        sig_in   : in  STD_LOGIC;
        sig_out  : out STD_LOGIC
    );
end debouncer;

architecture Behavioral of debouncer is
    constant WAIT_MS : integer := 20; -- 20ms ošetření
    constant WAIT_CYCLES : integer := WAIT_MS * 100000; 
    signal count : integer range 0 to WAIT_CYCLES := 0;
    signal sig_stable : std_logic := '0';
begin
    process(clk)
    begin
        if rising_edge(clk) then
            if rst = '1' then
                count <= 0;
                sig_stable <= '0';
            elsif sig_in /= sig_stable then
                if count < WAIT_CYCLES then
                    count <= count + 1;
                else
                    sig_stable <= sig_in;
                    count <= 0;
                end if;
            else
                count <= 0;
            end if;
        end if;
    end process;
    sig_out <= sig_stable;
end Behavioral;
