library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity top_safe is
    Port (
        CLK100MHZ : in  STD_LOGIC;
        BTNU      : in  STD_LOGIC; -- Reset
        BTND      : in  STD_LOGIC; -- Confirm
        SW        : in  STD_LOGIC_VECTOR(3 downto 0);
        CA, CB, CC, CD, CE, CF, CG : out STD_LOGIC;
        AN        : out STD_LOGIC_VECTOR(7 downto 0)
    );
end top_safe;

architecture Behavioral of top_safe is
    signal debounced_btnD, btnD_prev, confirm_pulse : std_logic;
    signal ce_mux : std_logic;
    signal state_sig : integer;
    signal digit_idx_sig : integer;
    signal seg_bus : std_logic_vector(6 downto 0);
begin

    -- Ošetření tlačítka
    deb_inst: entity work.debouncer
        port map (clk => CLK100MHZ, rst => BTNU, sig_in => BTND, sig_out => debounced_btnD);

    -- Detekce hrany (aby jeden stisk = jeden impuls)
    process(CLK100MHZ) begin
        if rising_edge(CLK100MHZ) then
            btnD_prev <= debounced_btnD;
            confirm_pulse <= debounced_btnD and (not btnD_prev);
        end if;
    end process;

    -- Generátor taktu pro displej
    clk_en_inst: entity work.clock_enable
        port map (clk => CLK100MHZ, rst => BTNU, ce_o => ce_mux);

    -- Logika trezoru
    fsm_inst: entity work.safe_fsm
        port map (clk => CLK100MHZ, rst => BTNU, btn_pulse => confirm_pulse, 
                  sw_i => SW, digit_idx => digit_idx_sig, state_o => state_sig);

    -- Driver displeje
    disp_inst: entity work.display_driver
        port map (clk => CLK100MHZ, ce => ce_mux, sw_i => SW, state_i => state_sig,
                  digit_idx => digit_idx_sig, seg_o => seg_bus, an_o => AN);

    -- Výstup na piny
    (CA, CB, CC, CD, CE, CF, CG) <= (seg_bus(6), seg_bus(5), seg_bus(4), seg_bus(3), seg_bus(2), seg_bus(1), seg_bus(0));

end Behavioral;
